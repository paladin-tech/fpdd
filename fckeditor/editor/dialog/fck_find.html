35 function ReplaceTextNodes( parentNode, regex, replaceValue, replaceAll )
36 {
37         for ( var i = 0 ; i < parentNode.childNodes.length ; i++ )
38         {
39                 var oNode = parentNode.childNodes[i] ;
40                 if ( oNode.nodeType == 3 )
41                 {
42                         var sReplaced = oNode.nodeValue.replace( regex, replaceValue ) ;
43                         if ( oNode.nodeValue != sReplaced )
44                         {
45                                 oNode.nodeValue = sReplaced ;
46                                 if ( ! replaceAll )
47                                         return true ;
48                         }
49                 }
50                 else
51                 {
52                         if ( ReplaceTextNodes( oNode, regex, replaceValue ) )
53                                 return true ;
54                 }
55         }
56         return false ;
57 }
